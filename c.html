O que é c++?

Em resumo, o C++ é uma linguagem de programação poderosa e versátil, usada para criar uma ampla variedade de aplicativos e sistemas, desde programas de desktop e jogos até sistemas críticos em tempo real e soluções de software de alto desempenho.

int p1 = 0;

int p2 = 0;

int res = 0;

void setup()
{
pinMode(A1, INPUT);
Serial.begin(9600);
pinMode(A0, INPUT);
pinMode(12, OUTPUT);

p1 = 0;
p2 = 0;
res = 0;
}

void loop()
{
p1 += analogRead(A1);
Serial.println("---");
Serial.println(p1);
p2 += analogRead(A0);
Serial.println("---");
Serial.println(p2);
if (p1 > p2) {
res += (p1 + p2);
Serial.println("---");
Serial.println(res);
digitalWrite(12, HIGH);
delay(1000); // Wait for 1000 millisecond(s)
} else {
digitalWrite(12, LOW);
}
delay(1000); // Wait for 1000 millisecond(s)
}
